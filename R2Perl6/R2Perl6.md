## R 语言中对应位置元素的最小值，最大值
> \>    a <- c(1,2,9,3,5)

> \>    b <- c(2,3,5,1,9)

> \>    c <- c(2,3,4,5,1)


> \> pmin(a,b,c)

> [1] 1 2 4 1 1

> \> pmax(a,b,c)

> [1] 2 3 9 5 9


> ## Perl 6 版：


> \> my @a = (1,2,9,3,5)

> \> my @b = (2,3,5,1,9)

> \> my @c = (2,3,4,5,1)


> \> ( @a>>min<<@b )>>min<< @c

> 1 2 4 1 1

> \> ( @a>>max<<@b )>>max<< @c

> 2 3 9 5 9


## 或者使用超运算符 [ Zmin ]、[ Zmax ]

> \> @a [Zmin] @b [Zmin] @c

> 1 2 4 1 1

> \> @a [Zmax] @b [Zmax] @c

> 2 3 9 5 9
